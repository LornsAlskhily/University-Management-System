
//use dbdiagram.io web

Table Users {
  id varchar [primary key]
  role enum 
  password varchar
  full_name varchar
}

Table Students{
  id varchar [primary key] 
  balance decimal
}

Table Finances{
  id varchar [primary key]
}

Table Courses{
  id varchar [primary key]
  name varchar 
  preRequisite_id varchar 
  major varchar
  hours integer
}

Table Division{
  id integer [primary key]
  Divsion_Number integer [unique]
  Course_id varchar 
  Capacity integer
  time datetime
  date datetime 
  lecturer_id varchar
}

Table Payment{
  id varchar [primary key]
  amount decimal
  status enum
  pay_date datetime
  resolved_date datetime
  student_id varchar
  Finance_id varchar
}


ref: Courses.preRequisite_id > Courses.id

ref: Division.Course_id > Courses.id

ref: Division.lecturer_id > Users.id

ref: Students.id - Users.id

ref : Finances.id - Users.id

ref:Payment.student_id > Students.id

ref: Payment.Finance_id > Finances.id
