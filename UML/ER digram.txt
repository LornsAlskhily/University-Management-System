
Table Users {
  id varchar [primary key]
  username varchar
  role enum 
  password varchar
}


Table Courses{
  id varchar [primary key]
  name varchar 
  preRequisite_id varchar 
  major varchar
  hours integer
}

Table Division{
  id integer [primary key]
  Courses_id varchar [primary key]
  Capacity integer
  time varchar
  date varchar 
  Lecturer varchar
}

ref: Courses.preRequisite_id > Courses.id

ref: Division.Courses_id > Courses.id

ref: Division.Lecturer > Users.id
